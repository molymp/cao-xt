function zeigToast(text, dauer = 2500) {
    const t = document.getElementById('toast');
    t.textContent = text;
    t.classList.add('sichtbar');
    setTimeout(() => t.classList.remove('sichtbar'), dauer);
}

async function apiFetch(url, optionen = {}) {
    try {
        const res = await fetch(url, {
            headers: { 'Content-Type': 'application/json' },
            ...optionen
        });
        const data = await res.json();
        if (!data.ok) {
            zeigToast('⚠️ ' + (data.fehler || 'Fehler'), 3000);
            return null;
        }
        return data;
    } catch (e) {
        zeigToast('⚠️ Verbindungsfehler', 3000);
        return null;
    }
}

// ── Bestätigungs-Overlay (ersetzt browser confirm()) ─────────
// Verwendung: zeigeBestaetigung('Text?', () => { /* bei Ja */ });
function zeigeBestaetigung(text, beiJa, jaText = 'Ja, fortfahren') {
    const overlay = document.getElementById('bestaetigung-overlay');
    document.getElementById('bestaetigung-text').textContent = text;
    document.getElementById('bestaetigung-ja').textContent = jaText;
    overlay.style.display = 'flex';

    // Einmalige Listener (werden nach Klick entfernt)
    const jaBtn   = document.getElementById('bestaetigung-ja');
    const neinBtn = document.getElementById('bestaetigung-nein');

    function schliessenUndJa() {
        overlay.style.display = 'none';
        jaBtn.removeEventListener('click', schliessenUndJa);
        neinBtn.removeEventListener('click', schliessenUndNein);
        beiJa();
    }
    function schliessenUndNein() {
        overlay.style.display = 'none';
        jaBtn.removeEventListener('click', schliessenUndJa);
        neinBtn.removeEventListener('click', schliessenUndNein);
    }

    jaBtn.addEventListener('click', schliessenUndJa);
    neinBtn.addEventListener('click', schliessenUndNein);
}

// Klick auf Hintergrund = Abbrechen
document.getElementById('bestaetigung-overlay').addEventListener('click', function(e) {
    if (e.target === this) {
        this.style.display = 'none';
    }
});

// Escape = Abbrechen (wird durch Seiten-eigene Escape-Handler nicht gestört,
// da dieser auf document registriert ist und zuerst feuert)
document.addEventListener('keydown', function(e) {
    if (e.key === 'Escape') {
        const overlay = document.getElementById('bestaetigung-overlay');
        if (overlay.style.display === 'flex') {
            overlay.style.display = 'none';
            e.stopImmediatePropagation(); // andere Escape-Handler nicht feuern
        }
    }
}, true); // capture phase
</script>
